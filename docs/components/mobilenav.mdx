slide-out navigation for mobile devices.

## Usage

```tsx
import { MobileNav } from "fromsrc/client";

<MobileNav title="docs" navigation={navigation} docs={docs} basePath="/docs" />;
```

## With logo

```tsx
<MobileNav
  title="fromsrc"
  logo={<Logo />}
  navigation={navigation}
  docs={docs}
/>
```

## With github

```tsx
<MobileNav
  title="docs"
  navigation={navigation}
  docs={docs}
  github="https://github.com/fromsrc/fromsrc"
/>
```

## Props

| prop       | type             | default | description          |
| ---------- | ---------------- | ------- | -------------------- |
| title      | string           | -       | header title         |
| logo       | ReactNode        | -       | logo element         |
| navigation | SidebarSection[] | -       | navigation structure |
| docs       | DocMeta[]        | -       | docs for search      |
| basePath   | string           | "/docs" | base path for links  |
| github     | string           | -       | github url           |
