Sidebar navigation link with automatic active state detection.

## Usage

```tsx
import { NavLink } from "fromsrc/client";

<NavLink href="/docs/intro">introduction</NavLink>;
```

## With icon

```tsx
<NavLink href="/docs/api" icon={<ApiIcon />}>
  api reference
</NavLink>
```

## With click handler

```tsx
<NavLink href="/docs/guide" onClick={closeMenu}>
  guide
</NavLink>
```

## Props

| prop     | type       | default | description      |
| -------- | ---------- | ------- | ---------------- |
| href     | string     | -       | Link destination |
| children | ReactNode  | -       | Link text        |
| icon     | ReactNode  | -       | Optional icon    |
| onClick  | () => void | -       | Click handler    |
