navigate between documentation sections.

## Usage

```tsx
import { TabNav } from "fromsrc/client"

const tabs = [
  { id: "guide", label: "guide", href: "/docs/guide" },
  { id: "api", label: "api", href: "/docs/api" },
  { id: "examples", label: "examples", href: "/docs/examples" },
]

<TabNav tabs={tabs} />
```

## With icons

```tsx
import { Book, Code, Layers } from "lucide-react"

const tabs = [
  { id: "guide", label: "guide", href: "/docs/guide", icon: <Book size={16} /> },
  { id: "api", label: "api", href: "/docs/api", icon: <Code size={16} /> },
  { id: "examples", label: "examples", href: "/docs/examples", icon: <Layers size={16} /> },
]

<TabNav tabs={tabs} />
```

## Custom matching

match specific paths:

```tsx
const tabs = [
  {
    id: "docs",
    label: "docs",
    href: "/docs",
    match: ["/docs/guide", "/docs/intro"],
  },
  {
    id: "api",
    label: "api",
    href: "/api",
    match: ["/api/hooks", "/api/components"],
  },
];
```

## Dropdown variant

for mobile or space-constrained layouts:

```tsx
<TabNavDropdown tabs={tabs} />
```

## Integration

place in sidebar header:

```tsx
import { Sidebar, TabNav } from "fromsrc/client";

function Layout({ children }) {
  return (
    <Sidebar header={<TabNav tabs={tabs} />} items={items}>
      {children}
    </Sidebar>
  );
}
```

## Props

### Tabnav

| prop | type     | description     |
| ---- | -------- | --------------- |
| tabs | NavTab[] | navigation tabs |

### Navtab

| prop  | type      | description                    |
| ----- | --------- | ------------------------------ |
| id    | string    | unique identifier              |
| label | string    | display label                  |
| href  | string    | target url                     |
| icon  | ReactNode | optional icon                  |
| match | string[]  | urls to match for active state |
