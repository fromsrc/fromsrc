documentation sidebar with search, navigation sections, and collapsible folders.

## usage

```tsx
import { Sidebar } from "fromsrc/client"

<Sidebar
  title="docs"
  navigation={navigation}
  github="https://github.com/org/repo"
/>
```

## navigation structure

### flat items

simple list of pages:

```tsx
const navigation = [
  {
    title: "getting started",
    items: docs.filter(d => d.slug?.startsWith("getting-started"))
  }
]
```

### collapsible folders

nested sections that can collapse:

```tsx
const navigation = [
  {
    title: "components",
    items: [
      {
        type: "folder",
        title: "layout",
        items: [
          { type: "item", title: "sidebar", href: "/docs/sidebar" },
          { type: "item", title: "header", href: "/docs/header" },
        ]
      },
      {
        type: "folder",
        title: "content",
        defaultOpen: true,
        items: [
          { type: "item", title: "callout", href: "/docs/callout" },
          { type: "item", title: "steps", href: "/docs/steps" },
        ]
      }
    ]
  }
]
```

### mixed

combine doc items with explicit items and folders:

```tsx
const navigation = [
  {
    title: "guides",
    items: [
      ...docs.filter(d => d.slug?.startsWith("guide")),
      { type: "item", title: "external", href: "https://example.com" },
      {
        type: "folder",
        title: "advanced",
        items: [
          { type: "item", title: "plugins", href: "/docs/plugins" },
        ]
      }
    ]
  }
]
```

## icons

add icons to items and folders:

```tsx
import { Book, Settings, Puzzle } from "lucide-react"

const navigation = [
  {
    title: "guides",
    items: [
      { type: "item", title: "intro", href: "/docs/intro", icon: <Book size={16} /> },
      {
        type: "folder",
        title: "config",
        icon: <Settings size={16} />,
        items: [
          { type: "item", title: "plugins", href: "/docs/plugins", icon: <Puzzle size={16} /> },
        ]
      }
    ]
  }
]
```

icons appear in the collapsed view and next to items in the expanded view.

## collapsible sidebar

enable icon-only collapsed mode with the `collapsible` prop:

```tsx
<Sidebar
  title="docs"
  navigation={navigation}
  collapsible
/>
```

when collapsed:
- shows only the toggle button
- navigation items show icons only
- expands on hover

## auto-expand folders

expand folders automatically to a certain depth:

```tsx
<Sidebar
  navigation={navigation}
  defaultOpenLevel={1}
/>
```

folders at depth 1 or less will be expanded by default.

## clickable folders

folders can also be links:

```tsx
{
  type: "folder",
  title: "api",
  href: "/docs/api",
  items: [
    { type: "item", title: "hooks", href: "/docs/api/hooks" },
    { type: "item", title: "content", href: "/docs/api/content" },
  ]
}
```

clicking the folder title navigates to the href. the expand/collapse button is separate.

## props

| prop | type | default | description |
|------|------|---------|-------------|
| title | string | required | site title |
| logo | ReactNode | - | site logo |
| navigation | SidebarSection[] | required | nav structure |
| basePath | string | "/docs" | base url path |
| github | string | - | github repo link |
| collapsible | boolean | false | enable collapse toggle |
| defaultOpenLevel | number | 0 | auto-expand folders to depth |

## types

```ts
interface SidebarItem {
  type: "item"
  title: string
  href: string
  icon?: ReactNode
}

interface SidebarFolder {
  type: "folder"
  title: string
  items: (SidebarItem | SidebarFolder)[]
  defaultOpen?: boolean
  icon?: ReactNode
  href?: string
}

interface SidebarSection {
  title: string
  items: (SidebarItem | SidebarFolder | DocMeta)[]
}
```
