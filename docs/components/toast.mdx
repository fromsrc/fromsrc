lightweight toast notifications for user feedback.

## setup

wrap your app with the toast provider:

```tsx
import { ToastProvider } from "fromsrc/client"

export default function Layout({ children }) {
  return (
    <ToastProvider>
      {children}
    </ToastProvider>
  )
}
```

## usage

use the `useToast` hook to trigger notifications:

```tsx
import { useToast } from "fromsrc/client"

function SaveButton() {
  const { add } = useToast()

  return (
    <button onClick={() => add({
      type: "success",
      message: "changes saved"
    })}>
      save
    </button>
  )
}
```

## types

### info

```tsx
add({ type: "info", message: "new version available" })
```

### success

```tsx
add({ type: "success", message: "file uploaded" })
```

### warning

```tsx
add({ type: "warning", message: "low storage" })
```

### error

```tsx
add({ type: "error", message: "connection failed" })
```

## custom duration

toasts auto-dismiss after 3 seconds by default:

```tsx
add({
  type: "info",
  message: "processing...",
  duration: 5000
})
```

## manual dismiss

use `remove` to dismiss a toast early:

```tsx
const { toasts, remove } = useToast()

toasts.map(toast => (
  <button onClick={() => remove(toast.id)}>dismiss</button>
))
```

## props

### toast

| prop | type | default | description |
|------|------|---------|-------------|
| message | string | required | notification text |
| type | "info" \| "success" \| "warning" \| "error" | required | toast style |
| duration | number | 3000 | auto-dismiss time in ms |

### useToast

| property | type | description |
|----------|------|-------------|
| toasts | Toast[] | current active toasts |
| add | (toast) => void | show a new toast |
| remove | (id) => void | dismiss a toast |
