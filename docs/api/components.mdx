## layout components

### Sidebar

navigation sidebar with sections and collapsible folders.

<TypeTable
  data={{
    title: { type: "string", required: true, description: "sidebar title text" },
    logo: { type: "ReactNode", description: "logo element next to title" },
    navigation: { type: "SidebarSection[]", required: true, description: "navigation structure" },
    basePath: { type: "string", default: '"/docs"', description: "base path for doc links" },
    github: { type: "string", description: "github repo url" },
    collapsible: { type: "boolean", description: "enable collapse to icon-only mode" },
    defaultOpenLevel: { type: "number", default: "0", description: "auto-open folder depth" },
  }}
/>

### SidebarSection

```typescript
interface SidebarSection {
  title: string
  items: (SidebarItem | SidebarFolder | DocMeta)[]
}
```

### SidebarItem

```typescript
interface SidebarItem {
  type: "item"
  title: string
  href: string
  icon?: ReactNode
}
```

### SidebarFolder

```typescript
interface SidebarFolder {
  type: "folder"
  title: string
  items: (SidebarItem | SidebarFolder)[]
  defaultOpen?: boolean
  icon?: ReactNode
  href?: string
}
```
