## Layout components

### Sidebar

Navigation sidebar with sections and collapsible folders.

<TypeTable
  data={{
    title: {
      type: "string",
      required: true,
      description: "Sidebar title text",
    },
    logo: { type: "ReactNode", description: "Logo element next to title" },
    navigation: {
      type: "SidebarSection[]",
      required: true,
      description: "Navigation structure",
    },
    basePath: {
      type: "string",
      default: '"/docs"',
      description: "Base path for doc links",
    },
    github: { type: "string", description: "GitHub repo URL" },
    collapsible: {
      type: "boolean",
      description: "Enable collapse to icon-only mode",
    },
    defaultOpenLevel: {
      type: "number",
      default: "0",
      description: "Auto-open folder depth",
    },
  }}
/>

### SidebarSection

```typescript
interface SidebarSection {
  title: string;
  items: (SidebarItem | SidebarFolder | DocMeta)[];
}
```

### SidebarItem

```typescript
interface SidebarItem {
  type: "item";
  title: string;
  href: string;
  icon?: ReactNode;
}
```

### SidebarFolder

```typescript
interface SidebarFolder {
  type: "folder";
  title: string;
  items: (SidebarItem | SidebarFolder)[];
  defaultOpen?: boolean;
  icon?: ReactNode;
  href?: string;
}
```
