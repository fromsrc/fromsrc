---
title: React Router
description: Configure fromsrc in React Router apps
---

Install dependencies:

```bash
bun add fromsrc react-router
```

Wire the adapter:

```tsx
import { AdapterProvider, reactRouterAdapter } from "fromsrc/react-router";
import type { ReactNode } from "react";

export function Providers({ children }: { children: ReactNode }) {
  return (
    <AdapterProvider adapter={reactRouterAdapter}>{children}</AdapterProvider>
  );
}
```

Mount in your app root:

```tsx
import { Providers } from "./providers";

export function AppRoot({ children }: { children: React.ReactNode }) {
  return <Providers>{children}</Providers>;
}
```

Use `fromsrc` content loaders in route modules for docs content and search indexing from `docs/`.
