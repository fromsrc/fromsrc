---
title: tanstack start
description: configure fromsrc in tanstack start apps
---

install dependencies:

```bash
bun add fromsrc @tanstack/react-router
```

wire the adapter:

```tsx
"use client";

import { AdapterProvider, tanstackAdapter } from "fromsrc/tanstack";
import type { ReactNode } from "react";

export function Providers({ children }: { children: ReactNode }) {
	return <AdapterProvider adapter={tanstackAdapter}>{children}</AdapterProvider>;
}
```

use the provider near your router provider so link/navigation components are consistent.

use the core content helpers from `fromsrc` for docs source parsing and search indexing.
