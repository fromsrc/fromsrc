---
title: Vite
description: Configure fromsrc in Vite apps
---

Install dependencies:

```bash
bun add fromsrc react-router
```

Wire the adapter with optional link override:

```tsx
import { AdapterProvider, createAdapter, viteAdapter } from "fromsrc/vite";
import { Link } from "react-router";
import type { ReactNode } from "react";

const adapter = createAdapter({
  ...viteAdapter,
  Link: ({ href, children }) => <Link to={href}>{children}</Link>,
});

export function Providers({ children }: { children: ReactNode }) {
  return <AdapterProvider adapter={adapter}>{children}</AdapterProvider>;
}
```

Use this provider once at app root and keep docs routes under `/docs/*`.
