---
title: astro
description: configure fromsrc in astro apps
---

install dependencies:

```bash
bun add fromsrc react react-dom
bun add -D astro @astrojs/react
```

enable the react integration:

```ts
import react from "@astrojs/react";
import { defineConfig } from "astro/config";

export default defineConfig({
	integrations: [react()],
});
```

wire the adapter in your react docs shell:

```tsx
import { AdapterProvider, astroAdapter } from "fromsrc/astro";
import type { ReactNode } from "react";

export function Providers({ children }: { children: ReactNode }) {
	return <AdapterProvider adapter={astroAdapter}>{children}</AdapterProvider>;
}
```

render this component in an astro island with `client:load` and keep markdown content in a shared `docs/` directory.
