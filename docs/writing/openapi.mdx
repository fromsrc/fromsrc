Render API reference pages directly from OpenAPI specifications.

## Quick usage

```mdx
<Openapi spec={spec} />
```

<Openapi
  spec={{
    openapi: "3.1.0",
    info: {
      title: "billing api",
      version: "1.0.0",
      description: "billing endpoints",
    },
    paths: {
      "/invoices": {
        get: {
          summary: "list invoices",
          tags: ["billing"],
          responses: {
            "200": {
              description: "ok",
              content: {
                "application/json": {
                  schema: {
                    type: "array",
                    items: {
                      type: "object",
                      properties: {
                        id: { type: "string" },
                        amount: { type: "number" },
                      },
                    },
                  },
                },
              },
            },
          },
        },
      },
    },
  }}
/>

## Filter endpoints

Filter by tag, method, or path substring:

```mdx
<Openapi spec={spec} tag="billing" />
<Openapi spec={spec} method="GET" />
<Openapi spec={spec} path="/invoices" />
<Openapi spec={spec} group="tag" />
```

## Import from JSON

For larger specs, load JSON and pass it to `Openapi`:

```tsx
import spec from "./openapi.json";

export default function Page() {
  return <Openapi spec={spec} />;
}
```

## Parser import

For parse-only flows, import the parser from the dedicated subpath:

```ts
import { parseOpenApi } from "fromsrc/openapi";

const parsed = parseOpenApi(spec);
```
