## create a new project

the fastest way to start is with create-fromsrc:

<Create />

<Steps>
<Step>

### run the command

this creates a new directory with everything configured.

```bash
bunx create-fromsrc
```

for non-interactive setup:

```bash
bunx create-fromsrc --name my-docs --framework next.js --yes
```

list available framework values:

```bash
bunx create-fromsrc --list
```

</Step>
<Step>

### enter the directory

```bash
cd my-docs
```

</Step>
<Step>

### start the dev server

```bash
bun run dev:up
bun run dev:status
bun run typecheck
```

for log tail and controlled restart:

```bash
bun run dev:tail 80
bun run dev:restart
bun run dev:down
```

</Step>
</Steps>

## next adapter

use the next.js adapter entry for framework wiring:

```tsx
import { AdapterProvider, nextAdapter } from "fromsrc/next"

export function Providers({ children }: { children: React.ReactNode }) {
  return <AdapterProvider adapter={nextAdapter}>{children}</AdapterProvider>
}
```

## react router adapter

use the react router adapter for browser-router apps:

```tsx
import { AdapterProvider, reactRouterAdapter } from "fromsrc/react-router"

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <AdapterProvider adapter={reactRouterAdapter}>
      {children}
    </AdapterProvider>
  )
}
```

## vite adapter

use the vite adapter for custom router stacks:

```tsx
import { AdapterProvider, createadapter, viteAdapter } from "fromsrc/vite"
import { Link } from "react-router"

const adapter = createadapter({
  ...viteAdapter,
  Link: ({ href, children }) => <Link to={href}>{children}</Link>,
})

export function Providers({ children }: { children: React.ReactNode }) {
  return <AdapterProvider adapter={adapter}>{children}</AdapterProvider>
}
```

## tanstack adapter

## troubleshooting

<Accordion>
  <AccordionItem title="cannot find module 'fromsrc'">
    ensure fromsrc is installed:
    ```bash
    bun add fromsrc
    ```
  </AccordionItem>
  <AccordionItem title="adapter not resolving">
    verify the subpath import matches your framework:
    ```tsx
    import { nextAdapter } from "fromsrc/next"
    import { reactRouterAdapter } from "fromsrc/react-router"
    import { viteAdapter } from "fromsrc/vite"
    import { tanstackAdapter } from "fromsrc/tanstack"
    import { remixAdapter } from "fromsrc/remix"
    import { astroAdapter } from "fromsrc/astro"
    import { parseOpenApi } from "fromsrc/openapi"
    import { createAlgoliaAdapter } from "fromsrc/algolia"
    import { createOramaAdapter } from "fromsrc/orama"
    ```
  </AccordionItem>
  <AccordionItem title="bundle size got bigger">
    run the tree-shake regression check:
    ```bash
    bun run check:shake
    ```
    this validates root, utility, and adapter imports stay inside budget.
  </AccordionItem>
  <AccordionItem title="search not working">
    search indexes content from your docs folder. ensure you have `.mdx` files with valid frontmatter:
    ```yaml
    ---
    title: page title
    description: page description
    ---
    ```
  </AccordionItem>
  <AccordionItem title="styles look wrong">
    import the fromsrc styles in your root layout or globals.css. check that tailwind is configured correctly.
  </AccordionItem>
  <AccordionItem title="pages not found">
    verify your catch-all route is at `app/docs/[[...slug]]/page.tsx` and imports from fromsrc correctly.
  </AccordionItem>
  <AccordionItem title="dev process stuck">
    restart and inspect logs:
    ```bash
    bun run dev:restart
    bun run dev:tail 80
    ```
  </AccordionItem>
</Accordion>
